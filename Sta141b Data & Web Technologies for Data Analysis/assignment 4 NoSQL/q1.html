  1. Consider all properties which have more than 100 reviews, what is their average price grouped by property type?
m$aggregate('[
            {"$match": {"number_of_reviews": {"$gt": 100}}}, 
            {"$group":
              {"_id":"$property_type", "avg_price":{"$avg":"$price"}}
            }
]') %>% rename(property_type=`_id`) %>% head(10)
